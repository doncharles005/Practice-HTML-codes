      <input type="password" id="regPassword" placeholder="Create a Password" required>

      <h3>Login</h3>

  <script>
    // Toggle between login and registration forms
    function toggleForms() {
      const registerForm = document.getElementById("registerForm");
      const loginForm = document.getElementById("loginForm");
      
      if (registerForm.style.display === "none") {
        registerForm.style.display = "block";
        loginForm.style.display = "none";
      } else {
        registerForm.style.display = "none";
        loginForm.style.display = "block";
      }
    }

    // Register a new user
    function registerUser() {
      const email = document.getElementById("regEmail").value.trim();
      const username = document.getElementById("regUsername").value.trim(); // Fixed: was regUserName
      const password = document.getElementById("regPassword").value.trim();

      if (!email || !username || !password) {
        document.getElementById("registerMessage").innerText = "All fields are required!";
        return;
      }

      // Correctly call Google Apps Script
      google.script.run
        .withSuccessHandler(function(response) {
          if (response === "success") {
            alert("You are registered successfully!");
            toggleForms(); // Switch to login form
          } else {
            document.getElementById("registerMessage").innerText = response;
          }
        })
        .withFailureHandler(function(error) {
          document.getElementById("registerMessage").innerText = "Registration failed. Please try again.";
          console.error("Registration error:", error);
        })
        .registerUser(email, username, password); // Ensure backend function name matches
    }

    // Login existing user
    function loginUser() {
      const email = document.getElementById("loginEmail").value.trim();
      const password = document.getElementById("loginPassword").value.trim();

      if (!email || !password) {
        document.getElementById("loginMessage").innerText = "Please enter email and password!";
        return;
      }

      google.script.run
        .withSuccessHandler(function(response) {
          // If response contains HTML (welcome page), inject it
          if (response && typeof response === 'string' && response.includes('<!DOCTYPE html')) {
            document.open();
            document.write(response);
            document.close();
          } else {
            // Otherwise, show error message
            document.getElementById("loginMessage").innerText = response;
          }
        })
        .withFailureHandler(function(error) {
          document.getElementById("loginMessage").innerText = "Error logging in. Please try again.";
          console.error("Login error:", error);
        })
        .loginUser(email, password); // Call server function
    }
  </script>
